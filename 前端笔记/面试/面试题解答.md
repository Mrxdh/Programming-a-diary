#### 关于this的面试题

```js
const o1 = {
	text: 'o1',
	fn: function () {
		return this.text
	}
}

const o2 = {
	text: 'o2',
	fn: function () {
		return o1.fn()
	}
}

const o3 = {
	text: 'o3',
	fn: function () {
		var fn = o1.fn
    // “裸奔”调用
		return fn()
	}
}
// o1
console.log(o1.fn())
// o1
console.log(o2.fn())
// undefined
console.log(o3.fn())
```

着重说一下第三个，o3的this指向的是window，所以调用结果是undefined





```js
const foo = {
	fn: function () {
		setTimeout(function () {
      // window
			console.log(this)
		})
	}
}
console.log(foo.fn())
```



```js
function foo (a) {
  this.a = a
}

const obj1 = {}

// 把foo的this做了改变
var bar = foo.bind(obj1)
// 修改了this为obj1的a对象
bar(2)
// 输出2
console.log(obj1.a)

var baz = new bar(3)
// 3
console.log(baz.a)
// 2
console.log(obj1.a)
```

  // 箭头函数的绑定无法修改
	return a => {

	}

	a: 2
}

const obj2 = {
	a: 3
}

const bar = foo.call(obj1)

console.log(bar.call(obj2))

#### 用代码实现一个New

```js
function newObject (Func, ...args) {
  /**
  	不用Object.create的替代方法
  	var obj = {}
  	obj.__proto__ = Function.prototype
  */
  // 绑定原型
	let obj = Object.create(Func.prototype)
  
  // 借用用外部的构造器传入对象和参数
  var result = Func.apply(obj, args)
  
  return typeof result === "object" ? result : obj
}
```

#### 请实现一个拥有on、emit、off的发布订阅

```javascript
class EmitEvent {
  constructor () {
    this.eventMap = {}
  }
  on (key, handle) {
    if (!this.eventMap[key]) {
      this.eventMap[key] = []
    }
    this.eventMap[key].push(handle)
  }
  
  emit (key, params) {
    if (!this.eventMap[key]) {
      return
    }
    
    this.eventMap[key].forEach(handle => {
      handle(params)
    })
  }
  
  off (key, handle) {
    this.eventMap[key].splice(this.eventMap[key].indexOf(handle), 1)
  }
}
```
